  def test_dirty?
    @attributes = {x: 1, y: 2}
    @record = TestModel.new(@attributes)
    assert_equal @record.dirty?, true
    @record.save
    assert_equal @record.dirty?, false
    @record2 = TestModel.create(@attributes)
    assert_equal @record2.dirty?, false
    @record2[:x]= 2
    assert_equal @record2.dirty?, true
  end
