    def retrieve(hash)
      result = hash.dup
      result.each_pair do |key, value|
        if value.is_a? RelationIdentifier
          klass = Kernel.const_get(value.class_name)
          if value.list? 
            result[key] = klass.find(hash[:id])
          else
            result[key] = klass.get(value.id)
          end
        else
          result[key] = value
        end          
      end
      result
    end
