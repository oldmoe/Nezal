      def save_one_to_one_relations
        self.class._one_to_one_hash.each_pair do | key, value |
          attribute = send(key)
          associated_attr = self.class._one_to_one_hash[key][:associated_attr]
          if !(value.class._one_to_one_hash.has_key? associated_attr)
            attribute.attributes[self.class._one_to_one_hash[key][:foreign_key]]= attributes[:id]
          end
          attribute.save if (attribute && attribute.dirty?)
          attributes[self.class._one_to_one_hash[key][:identified_by]] = attribute && attribute[:id]
        end
      end
